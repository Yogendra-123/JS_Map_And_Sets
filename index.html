Q .2 Problem: You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)
Input: abcadeecfb
Output: a=2 b=2 c=2 d=1 e=2 f=1 

            let s = 'AABBADCCS';
            let map = new Map();
            for(let i=0; i=s.length-1; i++){
            if(map.has(s[i])){
                let value = map.get(s[i]);
                map.set(s[i], value+1)
            }
            else{
                map.set(s[i],1)
            }
            }
            for (let [k,v] of map){
            console.log(k+ "=" + v)
            }
